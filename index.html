Integrasi Google Drive Pribadi Kirim Data Kunjungan Otomatis ke Google Sheets Anda

URL Web App Google Apps Script

Cara Memasang & Menghubungkan ke Email Google Pribadi Anda (Pembaruan Kode):

Jika Anda mendapatkan data kosong, hapus kode lama di Apps Script Anda dan gantikan dengan kode baru di bawah ini. Kode baru ini dirancang untuk menangkap parameter secara langsung tanpa terkendala CORS.

  1. Buka Google Apps Script menggunakan akun email pribadi Anda.
  2. Buka proyek yang telah Anda buat sebelumnya.
  3. Ganti isi berkas `Code.gs` seluruhnya dengan kode terbaru di bawah ini:
function doGet(e) {
  return handleData(e);
}

function doPost(e) {
  return handleData(e);
}

function handleData(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var p = e.parameter;
  
  // Jika parameter langsung kosong, coba periksa isi JSON postData
  if (!p.id && e.postData && e.postData.contents) {
    try {
      p = JSON.parse(e.postData.contents);
    } catch(err) {}
  }
  
  // Baris header kolom jika sheet masih kosong
  if (sheet.getLastRow() == 0) {
    sheet.appendRow(["ID Kunjungan", "Tanggal", "Nama Tamu", "WhatsApp", "Email", "Kategori", "Instansi", "Tujuan", "Metode", "Petugas", "Skor IKM", "Feedback", "Waktu Sinkronisasi"]);
  }
  
  if (p.id) {
    sheet.appendRow([
      p.id,
      p.datetime || p.rawDate || "",
      p.name || "",
      p.whatsapp || "",
      p.email || "",
      p.category || "",
      p.company || "",
      p.purpose || "",
      p.method || "",
      p.officer || "",
      p.rating || "",
      p.feedback || "",
      new Date().toLocaleString("id-ID")
    ]);
    return ContentService.createTextOutput(JSON.stringify({"status":"success", "message": "Data saved"}))
                         .setMimeType(ContentService.MimeType.JSON);
  } else {
    return ContentService.createTextOutput(JSON.stringify({"status":"error", "message": "No data received"}))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}
  1. Klik ikon Simpan (disket) di atas.
  2. SANGAT PENTING (Deploy Ulang): Klik Deploy (Terapkan) > Kelola penerapan (Manage deployments) > Klik ikon Pensil edit > Ubah versi ke "Versi baru" (New version) > Klik Terapkan (Deploy).
  3. Pastikan URL Web App yang Anda miliki adalah versi terbaru setelah Anda melakukan deploy ulang. Tempelkan kembali di kolom URL di atas dan klik **Simpan Setelan**.
